# 选项式api使用ts

<template>
  <div>
    <h1>{{count}}{{computedCount}}</h1>
    <button @click="handleClick(1)">count+1</button>
  </div>
</template>
<script lang="ts">
//选项式api需要使用defineComponent进行自动类型推断
import { defineComponent } from 'vue';
type Count = string | number;
interface obj {
  userName:string,
  age:number
}
export default defineComponent({
  data(){
    return{
      // 使用类型断言自定义数据类型
      count:0 as Count,
      list:[] as obj[]
    }
  },
  mounted() {
    this.count=123;
    this.list.push({userName:'xxx',age:18});
  },
  computed:{
    computedCount():number|string{
      //使用类型保护解决语法错误
      if(typeof this.count==='string'){
        return this.count
      } else {
        return this.count*2
      }
    }
  },
  methods:{
    handleClick(n:number):void{
      if(typeof this.count==='number'){
         this.count+=n
      }else {
        alert(this.count);
      }
    }
  },
  watch:{
    count(newV:string|number,oldV:string|number):void{
      console.log(newV,oldV)
    }
  }
})
</script>

# 选项式api组件通信使用ts

父->子
父组件
<template>
  <div>
    <Child :num="num" :list="list" :obj="obj"/>
  </div>
</template>
<script lang="ts">
import { defineComponent } from 'vue';
import Child from "@/views/Child.vue";
export default defineComponent({
  components:{
    Child
  },
  data(){
    return{
      num:1,
      list:[{userName:'xxx',age:18}],
      obj:{userName:'xxx'}
    }
  }
})
</script>

子组件
<template>
  <div>父组件传递的值{{num}}</div>
</template>
<script lang="ts">
import {defineComponent} from 'vue';
import type {PropType} from 'vue'
interface arr {
  userName:string,
  age:number
}
type Obj = {
  userName:string
}
export default defineComponent({
  name:'ChildComponent',
  props:{
    // 基本数据类型可通过vue内置类型进行限制
    num:[Number,String],
    // 引用数据类型需要通过PropType进行自定义
    list:Object as PropType<arr[]>,
    obj:Object as PropType<Obj>
  }
})
</script>

子->父
父组件
<template>
  <div>
    <Child @changeData="changeData"/>
  </div>
</template>
<script lang="ts">
import { defineComponent } from 'vue';
import Child from "@/views/Child.vue";
export default defineComponent({
  components:{
    Child
  },
  methods:{
    changeData(num:number):void{
      console.log(num)
    }
  }
})
</script>

子组件
<template>
  <div @click="changeParent">子组件</div>
</template>
<script lang="ts">
import {defineComponent} from 'vue';
export default defineComponent({
  name:'ChildComponent',
  // emits:['changeData']这种写法不能规定传递的参数类型
  // 该写法可以规定传递的参数类型，但必须返回一个值
  emits:{
    'changeData':(n:number)=>true
  },
  methods:{
    changeParent():void{
      this.$emit('changeData',1);
    }
  }
})
</script>

# 组合式api使用ts

<template>
    <div>{{ doubleNum}}</div>
    <button @click="handelClick">点击</button>
</template>
<script lang="ts" setup>
// 组合式api是自带类型注解的，当需要自定义类型注解时可以使用泛型
import { ref, reactive, computed } from 'vue';
let count = ref<number|string>(0);
count.value = 123;
count.value = 'xxx';

type T = {
    userName: string,
    age: number
}
let arr = reactive<T[]>([]);
arr.push({ userName: 'xxx', age: 18 });

let num = ref(1);
let doubleNum = computed(()=>{
    return num.value
})
let handelClick = ()=>{
    num.value = num.value*2
}
</script>

# 组合式api组件通信使用ts

父->子
父组件
<template>
    <Child :count="count" :list="list" />
</template>
<script lang="ts" setup>
import { ref, reactive } from 'vue';
import Child from './Child.vue';
let count = ref(0);
let list = reactive([{ userName: 'xxx', age: 18 }]);
</script>

子组件
<template>
    Child
</template>
<script lang="ts" setup>
import { defineProps } from 'vue';
// 第一种写法 Vue自带的props类型校验
// let props = defineProps({
//     count:Number
// })
// 第二种写法 使用ts进行类型校验
// interface Props {
//     count:number,
//     list:{userName:string,age:number}[]
// }
// let props =  defineProps<Props>();
// 获取到的参数在变量props中，是一个代理对象
// console.log(props);
</script>

子->父
父组件
<template>
    <Child @getData="getData" @getDataObj="getDataObj" />
</template>
<script lang="ts" setup>
import Child from './Child.vue';
interface DataObj {
    userName: string,
    age: number
}
let getData = (n: number) => {
    console.log(n);
}
let getDataObj = (n: DataObj) => {
    console.log(n);
}
</script>

子组件
<template>
    <button @click="handleClick">Child点击</button>
</template>
<script lang="ts" setup>
// 组合式api通过defineEmits触发自定义事件
import { defineEmits } from 'vue';
interface DataObj {
    userName: string,
    age: number
}
// 自定义事件名、参数类型、返回值类型定义。必须使用签名形式，否则报错
interface Emits {
    (e: 'getData', n: number): void,
    (e: 'getDataObj', n: DataObj): void,
}
let emit = defineEmits<Emits>();
let handleClick = () => {
    emit('getData', 1);
    emit('getDataObj', { userName: 'xxx', age: 18 });
}
</script>

# vuex封装（vue3+ts）

// 封装后会对state内的参数进行提示和类型校验
// 1、main.ts
import { createApp } from 'vue'
import App from './App.vue'
import router from './router'
import store, { key } from './store'

createApp(App).use(store, key).use(router).mount('#app')

// 2、store.ts
import { createStore, useStore as baseUseStore } from 'vuex'
import type { Store } from 'vuex'
import type { InjectionKey } from 'vue'
// 限制State中变量类型
export interface State {
  count: number
}
export const key: InjectionKey<Store<State>> = Symbol()
export function useStore() {
  return baseUseStore(key)
}
export default createStore<State>({
  state: {
    count: 0
  },
  getters: {
  },
  mutations: {
  },
  actions: {
  },
  modules: {
  }
})

// 3、使用页面
<template>
  <div></div>
</template>
<script lang="ts" setup>
import { useStore } from '@/store'
let store = useStore();
console.log(store.state.count);
</script>
