# sort排序

/**
 * 普通数组排序
 * @param {array} arr 要排序的数组
 * @param {string} sortType 升序或降序
 * @returns 排序后的数组
 */
function arrSort(arr, sortType = 'asc') {
    arr.sort((a, b) => {
        switch (sortType) {
            case 'asc':
                return a - b;
            case 'desc':
                return b - a;
        }
    });
    return arr
}

/**
 * 数组里面每一项为对象，需要对对象里面某个字段进行排序
 * @param {array} arr 要排序的数组
 * @param {string} field 根据该字段进行排序
 * @param {string} sortType 升序或降序
 * @returns 排序后的数组
 */
function arrSortByField(arr, field, sortType) {
    arr.sort((a, b) => {
        switch (sortType) {
            case 'asc':
                return a[field] - b[field];
            case 'desc':
                return b[field] - a[field];
        }
    })
    return arr
}

/**
 * 数组里面每一项为对象，需要对对象里面的时间字段进行排序
 * @param {array} arr 要排序的数组
 * @param {string} timeField 要排序的时间字段
 * @param {string} sortType 升序或降序
 * @returns 排序后的数组
 */
function arrSortByTime(arr, timeField, sortType) {
    arr.sort((a, b) => {
        switch (sortType) {
            case 'asc':
                return new Date(a[timeField]).getTime() - new Date(b[timeField]).getTime();
            case 'desc':
                return new Date(b[timeField]).getTime() - new Date(a[timeField]).getTime();
        }
    });
    return arr
}

# 快速排序

function quickSort(arr) {
    // 数组长度小于等于1直接返回数组
    if (arr.length <= 1) return arr;
    // 数组中间数下标
    let middleIndex = Math.ceil(arr.length / 2);
    // 数组中间数
    let middleNum = arr[middleIndex];
    // 将中间数剔除出去
    arr.splice(middleIndex, 1);
    // 值小于中间数的往leftArr中push
    let leftArr = [];
    // 值大于等于中间数的往rightArr中push
    let rightArr = [];
    let i = 0;
    while (i < arr.length) {
        if (arr[i] < middleNum) {
            leftArr.push(arr[i]);
        } else {
            rightArr.push(arr[i]);
        }
        i++
    };
    // 递归重复执行上面操作
    return [...quickSort(leftArr), middleNum, ...quickSort(rightArr)];
}

# 选择排序

function selectionSort(arr) {
    // 数组长度小于等于1直接返回数组
    if (arr.length <= 1) return arr;
    // 拿数组每一个和每一个的后一位进行对比
    for (let i = 0; i < arr.length - 1; i++) {
        for (let j = i + 1; j < arr.length; j++) {
            if (arr[i] > arr[j]) {
                // 将值大的放后面
                let temp = arr[i];
                arr[i] = arr[j];
                arr[j] = temp;
            }
        }
    }
    return arr
}

function removeDuplicate(array){
    let i = 0;
    let arr = [];
    while(i<array.length){
        if(!arr.includes(array[i])){
            arr.push(array[i])
        }
        i++
    }
    return arr
}

# base64转Buffer

function base64ToBuffer(dataURI) {
  var byteString = atob(dataURI.split(',')[1]);
  var buffer = new ArrayBuffer(byteString.length);
  var view = new Uint8Array(buffer);

  for (var i = 0; i < byteString.length; i++) {
    view[i] = byteString.charCodeAt(i);
  }

  return buffer;
}

# base64转Blob

function base64ToBlob(dataURI) {
  var mimeString = dataURI.split(',')[0].split(':')[1].split(';')[0];
  var buffer = base64ToBuffer(dataURI);
  return new Blob([buffer], {
    type: mimeString
  });
}

# 函数防抖

function debounce(fn, delay) {
  delay = delay || 1000;
  let timer = null;
  return function () {
    let context = this;
    let arg = arguments;
    if (timer) {
      clearTimeout(timer);
    }
    timer = setTimeout(() => {
      fn.apply(context, arg);
    }, delay);
  };
}

# 节流函数

function throttle(fn, delay = 300) {
  let timer = null;
  return function () {
    let context = this;
    let args = arguments;
    if (!timer) {
      timer = setTimeout(function () {
        fn.apply(context, args);
        clearTimeout(timer);
      }, delay);
    }
  };
}

# 复制文本

function copy(value, callback) {
  if (!document.queryCommandSupported('copy')) {
    callback('暂不支持复制')
    return
  }
  const textarea = document.createElement('textarea')
  textarea.value = value
  textarea.readOnly = Boolean('readOnly')
  document.body.appendChild(textarea)
  textarea.select()
  textarea.setSelectionRange(0, value.length)
  document.execCommand('copy')
  textarea.remove()
  callback('复制成功')
}
